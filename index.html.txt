  <script>
    // Passwort-Funktionen
    function checkPassword() {
      const password = document.getElementById('passwordInput').value;
      const correctPassword = '08102012';
      
      if (password === correctPassword) {
        document.getElementById('passwordOverlay').classList.add('hidden');
        document.getElementById('mainContent').classList.remove('hidden');
        // Canvas initialisieren nach dem Entsperren
        initializeCanvas();
      } else {
        document.getElementById('errorMessage').style.display = 'block';
        document.getElementById('passwordInput').value = '';
        document.getElementById('passwordInput').focus();
      }
    }

    function handlePasswordKeyPress(event) {
      if (event.key === 'Enter') {
        checkPassword();
      }
    }

    // Mini-Konfetti
    let canvas, ctx, confettiPieces = [], animationId;

    // Auto-Animation
    let carCanvas, carCtx, carAnimationId;
    let carX = -200;
    let carY = 0;

    function initializeCanvas() {
      canvas = document.getElementById('confetti');
      ctx = canvas.getContext('2d');
      carCanvas = document.getElementById('car');
      carCtx = carCanvas.getContext('2d');
      
      resizeCanvas();
    }

    function resizeCanvas() {
      if (canvas && carCanvas) {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
        carCanvas.width = window.innerWidth;
        carCanvas.height = window.innerHeight;
      }
    }
    window.addEventListener('resize', resizeCanvas);

    function random(min, max) { return Math.random() * (max - min) + min; }

    function createConfetti(count = 120) {
      confettiPieces = Array.from({ length: count }, () => ({
        x: random(0, canvas.width),
        y: random(-canvas.height, 0),
        size: random(6, 10),
        color: ['#7c3aed','#6d28d9','#10b981','#22c55e','#f59e0b','#ef4444'][Math.floor(random(0,6))],
        vx: random(-1, 1),
        vy: random(2, 4),
        rot: random(0, Math.PI * 2),
        vr: random(-0.1, 0.1)
      }));
    }

    function drawConfetti() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      confettiPieces.forEach(p => {
        p.x += p.vx;
        p.y += p.vy;
        p.rot += p.vr;

        if (p.y > canvas.height + 20) {
          p.y = random(-80, -20);
          p.x = random(0, canvas.width);
        }

        ctx.save();
        ctx.translate(p.x, p.y);
        ctx.rotate(p.rot);
        ctx.fillStyle = p.color;
        ctx.fillRect(-p.size/2, -p.size/2, p.size, p.size);
        ctx.restore();
      });
      animationId = requestAnimationFrame(drawConfetti);
    }

    function drawCar() {
      carCtx.clearRect(0, 0, carCanvas.width, carCanvas.height);
      
      // Schweizer Flagge (links)
      const chFlagX = 50;
      const chFlagY = carY + 50;
      const chFlagSize = 40;
      
      // Schweizer Flagge - rot
      carCtx.fillStyle = '#ff0000';
      carCtx.fillRect(chFlagX, chFlagY, chFlagSize, chFlagSize);
      
      // Schweizer Kreuz - wei√ü
      carCtx.fillStyle = '#ffffff';
      // Vertikaler Balken
      carCtx.fillRect(chFlagX + chFlagSize * 0.4, chFlagY + chFlagSize * 0.2, chFlagSize * 0.2, chFlagSize * 0.6);
      // Horizontaler Balken
      carCtx.fillRect(chFlagX + chFlagSize * 0.2, chFlagY + chFlagSize * 0.4, chFlagSize * 0.6, chFlagSize * 0.2);
      
      // Deutsche Flagge (rechts)
      const flagWidth = 60;
      const flagHeight = 40;
      const flagX = carCanvas.width - 110;
      const flagY = carY + 50;
      
      // Deutsche Flagge - schwarz
      carCtx.fillStyle = '#000000';
      carCtx.fillRect(flagX, flagY, flagWidth, flagHeight / 3);
      // Deutsche Flagge - rot
      carCtx.fillStyle = '#dd0000';
      carCtx.fillRect(flagX, flagY + flagHeight / 3, flagWidth, flagHeight / 3);
      // Deutsche Flagge - gold
      carCtx.fillStyle = '#ffce00';
      carCtx.fillRect(flagX, flagY + 2 * flagHeight / 3, flagWidth, flagHeight / 3);
      
      // Auto-K√∂rper (rot)
      carCtx.fillStyle = '#dc2626';
      carCtx.fillRect(carX, carY + 100, 120, 40);
      carCtx.fillRect(carX + 20, carY + 80, 80, 40);
      
      // Fenster (hellblau)
      carCtx.fillStyle = '#93c5fd';
      carCtx.fillRect(carX + 25, carY + 85, 30, 25);
      carCtx.fillRect(carX + 65, carY + 85, 30, 25);
      
      // R√§der (schwarz)
      carCtx.fillStyle = '#1f2937';
      carCtx.beginPath();
      carCtx.arc(carX + 30, carY + 140, 15, 0, Math.PI * 2);
      carCtx.fill();
      carCtx.beginPath();
      carCtx.arc(carX + 90, carY + 140, 15, 0, Math.PI * 2);
      carCtx.fill();
      
      // Scheinwerfer (gelb)
      carCtx.fillStyle = '#fbbf24';
      carCtx.fillRect(carX + 110, carY + 105, 8, 8);
      
      // Personen (K√∂pfe)
      carCtx.fillStyle = '#fbbf24';
      carCtx.beginPath();
      carCtx.arc(carX + 40, carY + 95, 8, 0, Math.PI * 2);
      carCtx.fill();
      carCtx.beginPath();
      carCtx.arc(carX + 80, carY + 95, 8, 0, Math.PI * 2);
      carCtx.fill();
      
      // Stra√üe
      carCtx.strokeStyle = '#6b7280';
      carCtx.lineWidth = 2;
      carCtx.setLineDash([20, 20]);
      carCtx.beginPath();
      carCtx.moveTo(0, carY + 160);
      carCtx.lineTo(carCanvas.width, carY + 160);
      carCtx.stroke();
      
      carX += 3;
      if (carX > carCanvas.width + 200) {
        carX = -200;
      }
      
      carAnimationId = requestAnimationFrame(drawCar);
    }

    function celebrate() {
      cancelAnimationFrame(animationId);
      cancelAnimationFrame(carAnimationId);
      createConfetti();
      drawConfetti();
      drawCar();
    }
